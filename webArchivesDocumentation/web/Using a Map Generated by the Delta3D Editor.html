Using a Map Generated by the Delta3D Editor   

Now that you have created a map from our previous tutorials, let’s now learn how to load it into a Delta3D scene.

First, let’s derive a class from the Delta3D Application class. 


#include <vector>

#include <dtDAL/map.h>
#include <dtDAL/project.h>
#include <dtDAL/actorproxy.h>

#include <dtABC/application.h>

#include <dtCore/walkmotionmodel.h>
#include <dtCore/refptr.h>
#include <dtCore/object.h>
#include <dtCore/camera.h>
#include <dtCore/transform.h>
#include <dtCore/globals.h>


class TestApp : public dtABC::Application
{
    public:
        TestApp() : dtABC::Application(“config.xml”)
	{
  	}
	virtual ~TestApp() 
	{
	}
};
Now that you have your application class set up, the next step is to set up a motion model to your camera so you can move around in the map and configure Delta3D. This is done with a few lines of code in the constructor and the addition of a member variable for our motion model. 

    public:
        TestApp()
	{
	    mWMM = new dtCore::WalkMotionModel(GetKeyboard(), GetMouse());
	    mWMM->SetTarget(GetCamera());

	}
    private:
            dtCore::RefPtr<dtCore::WalkMotionModel> mWMM;
Now you are all set up to move about in the map, so let’s load it in! The first thing you need to do is set your project context, just like you did in the editor from the previous tutorial. Next, you need to procure a reference to your map by passing the function the name of the map. Lastly, you load the map into the scene. This process is so simple it can also be completely done in the constructor of your TestApp class. 

    public:
        TestApp()
	{
	    mWMM = new dtCore::WalkMotionModel(GetKeyboard(), GetMouse());
	    mWMM->SetTarget(GetCamera());

	    dtDAL::Project::GetInstance().SetContext("MyProject");

	    dtDAL::Map &myMap = dtDAL::Project::GetInstance().GetMap("MyCoolMap");

	    dtDAL::Project::GetInstance().LoadMapIntoScene(myMap, *GetScene());
	}
The parameters of LoadMapIntoScene are as follows: a reference to the map you want to load, the scene to load the map into, and true to load in billboards, false to ignore them. This parameter is defaulted to false. Now let’s do something with the map. For simplicity, let’s make the helicopter fly across the screen. First you have to query the map for its proxies. Then you must process through the proxies until you find the correct one you are looking for and cast it. The Map class has a function called FindProxies, which takes a std::vector and a string as parameters. This function will fill the vector with the proxies from the map that invoked it with the proxies that have names matching the string specified. For example, parameter of “*” would return all the proxies in the map. Then, you can manipulate the actor however you want! The following code should go in the constructor of the TestApp class and the member variable “helicopter” should be declared in the private section of the class. 

	TestApp()
	{
	    // below the code above 
		mHelicopter = NULL;
		mWMM = new dtCore::WalkMotionModel(GetKeyboard(), GetMouse());
		mWMM->SetTarget(GetCamera());

		dtDAL::Project::GetInstance().SetContext("MyProject");

		dtDAL::Map &myMap = dtDAL::Project::GetInstance().GetMap("MyCoolMap");

		dtDAL::Project::GetInstance().LoadMapIntoScene(myMap, *GetScene());

                typedef std::vector< dtCore::RefPtr<dtDAL::ActorProxy> > ProxyVector;

                ProxyVector proxies;
	        myMap.FindProxies(proxies, "Static*");

       
                ProxyVector::iterator end_iter = proxies.end();

                // process the proxies
                for (ProxyVector::iterator iter = proxies.begin(); iter != end_iter; ++iter)
                {
                  if ((*iter)->GetName() == "StaticMesh0")
                  {
                    mHelicopter = dynamic_cast<dtCore::Object*>((*iter)->GetActor());
                  }
                }
		
                if (mHelicopter == NULL)
                {	
          	  return;
                }
          }
	// Since we need to change the transform of the helicopter every frame, 
	// we should override the OnMessage function and do it there.
        virtual void OnMessage(MessageData *data)
        {
          static double step = 0.0f;

          if(data->message == "preframe")
          {
            double dt = *static_cast<double*>(data->userData);
            step += dt;
 
	    dtCore::Transform x(0, step, 0, 0, 0, 0);
 
            if(mHelicopter != NULL)
            {
              mHelicopter->SetTransform(x);
            }
          }
	}

        private:
	   dtCore::RefPtr<dtCore::WalkMotionModel> mWMM;
	   dtCore::RefPtr<dtCore::Object> mHelicopter;
That’s it. Now all you need is a main function and you are ready to go. The final file looks something like this: 

#include <vector>

#include <dtDAL/map.h>
#include <dtDAL/project.h>
#include <dtDAL/actorproxy.h>

#include <dtABC/application.h>

#include <dtCore/walkmotionmodel.h>
#include <dtCore/refptr.h>
#include <dtCore/object.h>
#include <dtCore/camera.h>
#include <dtCore/transform.h>
#include <dtCore/globals.h>


class TestApp : public dtABC::Application
{
  private:
	dtCore::RefPtr<dtCore::WalkMotionModel> mWMM;
	dtCore::RefPtr<dtCore::Object> mHelicopter;

  public:
   TestApp()
   : Application("config.xml")
   {
      mHelicopter = NULL;
      mWMM = new dtCore::WalkMotionModel(GetKeyboard(), GetMouse());
      mWMM->SetTarget(GetCamera());

      dtDAL::Project::GetInstance().SetContext("MyProject");

      dtDAL::Map &myMap = dtDAL::Project::GetInstance().GetMap("MyCoolMap");

      dtDAL::Project::GetInstance().LoadMapIntoScene(myMap, *GetScene());

      typedef std::vector< dtCore::RefPtr<dtDAL::ActorProxy> > ProxyVector;

      ProxyVector proxies;
      myMap.FindProxies(proxies, "Static*");

       
      ProxyVector::iterator end_iter = proxies.end();

      // process the proxies
      for (ProxyVector::iterator iter = proxies.begin(); iter != end_iter; ++iter)
      {
	   if ((*iter)->GetName() == "StaticMesh0")
         {
            mHelicopter = dynamic_cast<dtCore::Object*>((*iter)->GetActor());
         }
      }
		
      if (mHelicopter == NULL)
      {	
         return;
      }
   }
   virtual void OnMessage(MessageData *data)
   {
      static double step = 0.0f;

      if(data->message == "preframe")
      {
         double dt = *static_cast<double*>(data->userData);
         step += dt;

	 dtCore::Transform x(0, step, 0, 0, 0, 0);

         if(mHelicopter != NULL)
         {
            mHelicopter->SetTransform(x);
         }
      }
   }
   
   virtual ~TestApp() 
   {

   }

};

int main()
{
    // Setup the data file search paths for the config file and the models files.
    // This is best done in main prior to configuring app. That way the paths
    // are ensured to be correct when loading data.
    dtCore::SetDataFilePathList( ".;" + dtCore::GetDeltaDataPathList() );

    dtCore::RefPtr<TestApp> app = new TestApp();
    app->Config();
    app->Run();
    return 0;
}
